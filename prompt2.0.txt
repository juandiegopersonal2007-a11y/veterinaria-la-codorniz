prisma// schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model User {
  id        String   @id @default(cuid())
  email     String   @unique
  password  String
  name      String
  role      Role     @default(ADMIN)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

model Client {
  id        String   @id @default(cuid())
  name      String
  phone     String
  email     String?
  address   String?
  pets      Pet[]
  appointments Appointment[]
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt
}

model Pet {
  id          String       @id @default(cuid())
  name        String
  species     String
  breed       String?
  birthDate   DateTime?
  weight      Float?
  chipNumber  String?      @unique
  photoUrl    String?
  clientId    String
  client      Client       @relation(fields: [clientId], references: [id], onDelete: Cascade)
  medicalHistory MedicalRecord[]
  appointments   Appointment[]
  createdAt   DateTime     @default(now())
  updatedAt   DateTime     @updatedAt
}

model MedicalRecord {
  id          String   @id @default(cuid())
  petId       String
  pet         Pet      @relation(fields: [petId], references: [id], onDelete: Cascade)
  date        DateTime @default(now())
  diagnosis   String
  treatment   String
  notes       String?
  vetName     String
  createdAt   DateTime @default(now())
}

model Appointment {
  id        String            @id @default(cuid())
  clientId  String
  client    Client            @relation(fields: [clientId], references: [id], onDelete: Cascade)
  petId     String
  pet       Pet               @relation(fields: [petId], references: [id], onDelete: Cascade)
  date      DateTime
  service   ServiceType
  status    AppointmentStatus @default(PENDING)
  notes     String?
  createdAt DateTime          @default(now())
  updatedAt DateTime          @updatedAt
}

model Service {
  id          String  @id @default(cuid())
  name        String
  description String
  price       Float
  imageUrl    String?
  active      Boolean @default(true)
}

enum Role {
  ADMIN
  STAFF
}

enum ServiceType {
  CONSULTA
  ESTETICA
  BANO
  CIRUGIA_MENOR
  VACUNACION
  DESPARASITACION
  OTRO
}

enum AppointmentStatus {
  PENDING
  CONFIRMED
  COMPLETED
  CANCELLED
}

Reglas de seguridad

Todos los endpoints excepto GET /api/pets/chip/:chipNumber y POST /api/auth/login requieren JWT válido
Validar todos los inputs con Zod
Sanitizar contra XSS con xss-clean
Rate limiting con express-rate-limit (100 req/15min por IP)
Helmet.js para headers de seguridad
CORS restringido al dominio del frontend
Contraseñas hasheadas con bcrypt (salt rounds: 12)
JWT con expiración de 8h, refresh token de 7d
Logs de errores con Winston

Protección de rutas admin
typescript// middleware.ts
import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'

export function middleware(request: NextRequest) {
  const token = request.cookies.get('auth-token')
  if (!token && request.nextUrl.pathname.startsWith('/admin')) {
    return NextResponse.redirect(new URL('/login', request.url))
  }
  return NextResponse.next()
}

export const config = {
  matcher: ['/admin/:path*']
}

🚀 CI/CD — GitHub Actions
Deploy Frontend (Vercel)
yaml# .github/workflows/deploy-web.yml
name: Deploy Web

on:
  push:
    branches: [main]
    paths: ['apps/web/**']

jobs:
  deploy:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - uses: actions/setup-node@v4
        with:
          node-version: 20
      - run: npm ci
      - run: npm run build --workspace=apps/web
      - uses: amondnet/vercel-action@v25
        with:
          vercel-token: ${{ secrets.VERCEL_TOKEN }}
          vercel-org-id: ${{ secrets.VERCEL_ORG_ID }}
          vercel-project-id: ${{ secrets.VERCEL_PROJECT_ID }}
          vercel-args: '--prod'
